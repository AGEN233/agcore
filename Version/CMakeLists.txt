execute_process(
    COMMAND git -C "${CMAKE_CURRENT_LIST_DIR}" rev-parse --short HEAD
    OUTPUT_VARIABLE AGCORE_GIT_HASH
    OUTPUT_STRIP_TRAILING_WHITESPACE
    ERROR_QUIET
    RESULT_VARIABLE AGCORE_GIT_HASH_RESULT
)
if(NOT AGCORE_GIT_HASH_RESULT EQUAL 0 OR AGCORE_GIT_HASH STREQUAL "")
    set(AGCORE_GIT_HASH "unknown")
endif()

execute_process(
    COMMAND git -C "${CMAKE_CURRENT_LIST_DIR}" branch --show-current
    OUTPUT_VARIABLE AGCORE_GIT_BRANCH
    OUTPUT_STRIP_TRAILING_WHITESPACE
    ERROR_QUIET
    RESULT_VARIABLE AGCORE_GIT_BRANCH_RESULT
)
if(NOT AGCORE_GIT_BRANCH_RESULT EQUAL 0 OR AGCORE_GIT_BRANCH STREQUAL "")
    set(AGCORE_GIT_BRANCH "unknown")
endif()

string(TIMESTAMP AGCORE_BUILD_TIME "%Y-%m-%d %H:%M:%S")

set(AGCORE_VERSION_GENERATED_DIR "${CMAKE_CURRENT_BINARY_DIR}/generated")
file(MAKE_DIRECTORY "${AGCORE_VERSION_GENERATED_DIR}")
configure_file(
    "${CMAKE_CURRENT_LIST_DIR}/agcore_version.h"
    "${AGCORE_VERSION_GENERATED_DIR}/agcore_version.h"
    @ONLY
)

target_sources(${COMPONENT_LIB} PRIVATE
    "agcore_version.c"
)

target_include_directories(${COMPONENT_LIB} PUBLIC
    "${AGCORE_VERSION_GENERATED_DIR}"
    "."
)
